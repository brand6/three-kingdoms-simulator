```gdscript
func get_xun_label(year: int, month: int, xun: int) -> String:
	if year <= 0 or month <= 0 or xun <= 0:
		return "—"
	return "%d年 / %d月 / 第%d旬" % [year, month, xun]

func get_current_label() -> String:
```
